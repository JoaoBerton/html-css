class Planeta:
    def __init__(self, nome, massa, distancia_do_sol):
        self.nome = nome
        self.massa = massa  # em quilogramas
        self.distancia_do_sol = distancia_do_sol  # em milhões de quilômetros

    def __str__(self):
        return f"Planeta: {self.nome}, Massa: {self.massa} kg, Distância do Sol: {self.distancia_do_sol} milhões de km"
